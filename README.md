# dragoncompiler

```

# DRAGON C CODE SAMPLE

# allocates 5 bytes in ram under label "a".
vec a : 5; # "a" is now a global symbol

var b = 9; # allocates 1 byte for variable 'b' with value 9

rom mynumbers : 10, 20, 32, 10, 6, 4, 2; # stores this in rom

func simple() { # func_simple:
  var b;     # SP -= 1;
  var c;     # SP -= 1;
  b = 68;    # LBI RX, 68; SB RX, +2(SP)
  c = 4;     # LBI RX, 4;  SB RX, +1(SP)
  b = b + c;

  return 7;
  # LBI RF, 7

  # release stack # SP += 2
  # JRT
}

func main(){
  var a, d;
  a = 10;
  d = simple();

  CALL simple;
  LB RF, +1(SP)

  a = a + d;
  PRINT(a);

  forever{
    a++;
    if ( a == 150 ){
        break
    }
    DRAW(0, 0, 14, 14, 3);
    HALT();
  }
}
```

## Grammar Rules ( WIP )
```
BLOCK :
	CURLY_BRACKET_OPEN
	STATEMENTS
	CURLY_BRACKET_CLOSED

STATEMENTS :
  STATEMENT
  STATEMENTS STATEMENT

STATEMENT :
	VAR_DECLARATION
	ROM_DECLARATION
	VEC_DECLARATION
	VAR_DECLARATION_AND_INIT
	VAR_ASIGNMENT
	RETURN_STATEMENT
  FUNC_CALL
  IF

EXPR :
  GENERAL_IDENTIFIER
  NUMERAL
  EXPR PLUS EXPR
  EXPR XOR EXPR

VAR_DECLARATION :
	KEYWORD_VAR GENERAL_IDENTIFIER SEMICOLON

NUMERAL_LIST :
  NUMERAL COMMA NUMERAL
  NUMERAL_LIST COMMA NUMERAL

ROM_DECLARATION :
  KEYWORD_ROM GENERAL_IDENTIFIER COLON NUMERAL SEMICOLON
	KEYWORD_ROM GENERAL_IDENTIFIER COLON NUMERAL_LIST SEMICOLON

VEC_DECLARATION :
	KEYWORD_VEC GENERAL_IDENTIFIER COLON NUMERAL SEMICOLON

PARAM_LIST :
  GENERAL_IDENTIFIER COMMA GENERAL_IDENTIFIER
  PARAM_LIST COMMA GENERAL_IDENTIFIER

FUNC_DECLARATION :
	KEYWORD_FUNC GENERAL_IDENTIFIER PARATHESIS_OPEN PARANTHESIS_CLOSED  BLOCK
  KEYWORD_FUNC GENERAL_IDENTIFIER PARATHESIS_OPEN GENERAL_IDENTIFIER PARANTHESIS_CLOSED  BLOCK
  KEYWORD_FUNC GENERAL_IDENTIFIER PARATHESIS_OPEN PARAM_LIST PARANTHESIS_CLOSED  BLOCK

VAR_DECLARATION_AND_INIT :
	KEYWORD_VAR GENERAL_IDENTIFIER EQUALS NUMERAL SEMICOLON
	KEYWORD_VAR GENERAL_IDENTIFIER EQUALS GENERAL_IDENTIFIER SEMICOLON

VAR_ASIGNMENT :
	GENERAL_IDENTIFIER EQUALS GENERAL_IDENTIFIER SEMICOLON
	GENERAL_IDENTIFIER EQUALS NUMERAL SEMICOLON

RETURN_STATEMENT :
	KEYWORD_RETURN GENERAL_IDENTIFIER SEMICOLON
	KEYWORD_RETURN EQUALS SEMICOLON

FUNC_CALL :
  GENERAL_IDENTIFIER FUNC_PARAM_LIST SEMICOLON

CONDITION :
  EXPR LOP_EQUALS EXPR
  EXPR LOP_NOT_EQUALS EXPR
  EXPR LOP_GREATER_THAN EXPR
  EXPR LOP_GREATER_EQUAL EXPR
  EXPR LOP_LESS_THAN EXPR
  EXPR LOP_LESS_EQUAL EXPR

IF :
  KEYWORD_IF PARATHESIS_OPEN CONDITION PARANTHESIS_CLOSED BLOCK

FOREVER :
  KEYWORD_FOREVER BLOCK

BREAK_STATEMENT :
  KEYWORD_BREAK SEMICOLON;
```
